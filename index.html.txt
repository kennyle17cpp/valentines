<script>
  const noButton = document.getElementById('no');
  const container = document.querySelector('.buttons');

  // On touch or click, move the "No" button randomly
  noButton.addEventListener('touchstart', moveNo);
  noButton.addEventListener('click', moveNo);

  function moveNo() {
    const containerWidth = container.offsetWidth - noButton.offsetWidth;
    const containerHeight = container.offsetHeight - noButton.offsetHeight;

    const randomX = Math.floor(Math.random() * containerWidth);
    const randomY = Math.floor(Math.random() * containerHeight);

    noButton.style.left = randomX + 'px';
    noButton.style.top = randomY + 'px';
  }

  // Heart background animation
  function createHeart() {
    const heart = document.createElement('div');
    heart.classList.add('heart');
    heart.style.left = Math.random() * window.innerWidth + 'px';
    heart.style.animationDuration = (2 + Math.random() * 3) + 's';
    document.body.appendChild(heart);
    setTimeout(() => heart.remove(), 4000);
  }

  setInterval(createHeart, 500);

  // Yes click with confetti
  function yesClicked() {
    for(let i = 0; i < 30; i++) {
      const confetti = document.createElement('div');
      confetti.classList.add('confetti');
      confetti.style.left = Math.random() * window.innerWidth + 'px';
      confetti.style.backgroundColor = `hsl(${Math.random() * 360}, 100%, 50%)`;
      document.body.appendChild(confetti);
      setTimeout(() => confetti.remove(), 2000);
    }
    alert("Yay! I knew you would say yes, HongAnh! ‚ù§Ô∏è");
  }
</script>
